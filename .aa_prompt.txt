<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;

class RankingController extends Controller
{
    public function getRanking(Request $request)
    {
        $validated = $request->validate([
            'sort_by' => 'in:recovery_rate,hit_rate,total_bets,follower_count', 
        ]);

        $sortBy = $validated['sort_by'] ?? 'recovery_rate';
        $daily_ranking_info = $this->fetchRankingData('daily', $sortBy);
        $weekly_ranking_info = $this->fetchRankingData('weekly', $sortBy);
        $monthly_ranking_info = $this->fetchRankingData('monthly', $sortBy);

        return view('ranking', compact('daily_ranking_info', 'weekly_ranking_info', 'monthly_ranking_info', 'sortBy'));
    }

    private function fetchRankingData($period, $sortBy)
    {
        $viewName = "{$period}_performance_summary_view";
        $rankingData = $this->getRankingQuery($viewName, $sortBy)->get();

        return $this->formatRankingData($rankingData, $viewName);
    }

    private function getRankingQuery($viewName, $sortBy)
    {
        return DB::table($viewName)
            ->join('users', "{$viewName}.user_id", '=', 'users.UserID')
            ->leftJoin('follows', 'users.UserID', '=', 'follows.FolloweeID')
            ->select(
                'users.UserID',
                'users.name as username', 
                DB::raw("CONCAT('/', users.IconImageURL) as icon_url"), 
                DB::raw("CAST({$viewName}.recovery_rate AS UNSIGNED) as recovery_rate"),
                DB::raw("CAST({$viewName}.hit_rate AS UNSIGNED) as hit_rate"),
                "{$viewName}.total_bets",
                DB::raw('COUNT(DISTINCT follows.FollowerID) as follower_count')
            )
            ->groupBy('users.UserID', 'users.name', 'users.IconImageURL', "{$viewName}.recovery_rate", "{$viewName}.hit_rate", "{$viewName}.total_bets")
            ->orderBy($sortBy, 'desc')
            ->orderBy('total_bets', 'desc')
            ->take(3);
    }

    private function formatRankingData($rankingData, $viewName)
    {
        return $rankingData->map(function ($user) {
            $articles = $this->getUserArticles($user->UserID);

            return [
                'username' => $user->username,
                'icon_url' => $user->icon_url,
                'recovery_rate' => $user->recovery_rate,
                'hit_rate' => $user->hit_rate,
                'total_bets' => $user->total_bets,
                'follower_count' => $user->follower_count,
                'articles' => $articles
            ];
        })->toArray();
    }

    private function getUserArticles($userId)
    {
        return DB::table('articles')
            ->select('articles.ArticleID as article_id', 'articles.Title as article_title', 'articles.Price as price', 'articles.PublishedAt as creation_date', 'users.name as author', 'articles.TopImageURL', 'articles.UserID')
            ->leftJoin('users', 'articles.UserID', '=', 'users.UserID')
            ->where('articles.UserID', $userId)
            ->where('articles.IsPublished', true)
            ->orderBy('articles.PublishedAt', 'desc')
            ->take(2)
            ->get()
            ->map(function ($article) {
                return [
                    'article_id' => $article->article_id,
                    'article_title' => $article->article_title,
                    'price' => $article->price,
                    'creation_date' => $article->creation_date instanceof \DateTime ? $article->creation_date->format('Ymd') : $article->creation_date,
                    'top_image_url' => $article->TopImageURL
                ];
            })
            ->toArray();
    }
}

<?php

namespace App\Http\Controllers;

use App\Models\User;
use App\Models\Article;
use App\Models\Race;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Auth;

class HomeController extends Controller
{
    public function index()
    {
        // ログイン状態と認証状態のチェック
        if (Auth::check() && Auth::user()->hasVerifiedEmail()) {
            $is_logged_in = true;
        } else {
            $is_logged_in = false;
        }

        // 注目レース情報
        $focus_race_info = [
            [
                "race_title" => "JRA70周年記念 ラジオNIKKEI賞（GⅢ）",
                "race_link" => "https://example.com/races/tokyo"
            ]
        ];

        $header_image_1 = "/admin_images/header1";
        $header_image_2 = "/admin_images/header2";

        $ranking_info = $this->getRankingInfo();

        $new_article_info = $this->getNewArticleInfo();

        $recent_race_info = $this->getRecentRaceInfo();

        // ログインユーザーのアイコン情報を追加
        $userIcon = Auth::check() ? Auth::user()->IconImageURL : 'user_images/icon_default';

        return view('home', compact(
            'is_logged_in',
            'focus_race_info',
            'header_image_1',
            'header_image_2',
            'ranking_info',
            'recent_race_info',
            'new_article_info',
            'userIcon'
        ));
    }

    private function getRankingInfo()
    {
        return User::select(
            DB::raw("COALESCE(users.IconImageURL, 'user_images/icon_default') as user_icon"),
            'users.name as user_name', 
            DB::raw('COUNT(follows.FollowerID) as followers'),
            'daily_performance_summary_view.recovery_rate',
            'daily_performance_summary_view.total_bets',
            'daily_performance_summary_view.hit_rate',
            'users.created_at'
        )
            ->leftJoin('follows', 'users.UserID', '=', 'follows.FollowerID')
            ->leftJoin('daily_performance_summary_view', 'users.UserID', '=', 'daily_performance_summary_view.user_id')
            ->groupBy(
                'users.UserID', 
                'users.IconImageURL', 
                'users.name', 
                'daily_performance_summary_view.recovery_rate',
                'daily_performance_summary_view.total_bets',
                'daily_performance_summary_view.hit_rate',
                'users.created_at'
            )
            ->orderByDesc('recovery_rate')
            ->orderByDesc('total_bets')
            ->orderByDesc('hit_rate')
            ->orderBy('users.created_at')
            ->limit(3)
            ->get()
            ->toArray();
    }

    private function getNewArticleInfo()
    {
        return Article::select(
            'articles.Title as title',
            'articles.TopImageURL as header_image',
            DB::raw('COUNT(likes.LikeID) as likes'),
            'articles.Price as price',
            'articles.PublishedAt as creation_date',
            DB::raw("COALESCE(users.IconImageURL, 'user_images/icon_default') as user_icon"),
            'users.name as user_name'
        )
            ->leftJoin('users', 'articles.UserID', '=', 'users.UserID')
            ->leftJoin('likes', 'articles.ArticleID', '=', 'likes.ArticleID')
            ->groupBy('articles.ArticleID', 'users.UserID')
            ->orderByDesc('articles.PublishedAt')
            ->limit(8)
            ->get()
            ->toArray();
    }

    private function getRecentRaceInfo()
    {
        return Article::select(
            'races.RaceDateTime as race_date',
            'articles.Title as title',
            DB::raw('COUNT(likes.LikeID) as likes'),
            'articles.Price as price',
            'articles.PublishedAt as creation_date',
            DB::raw("COALESCE(users.IconImageURL, 'user_images/icon_default') as user_icon"),
            'users.name as user_name',
            'articles.TopImageURL as header_image'
        )
            ->leftJoin('races', 'articles.RaceID', '=', 'races.RaceID')
            ->leftJoin('users', 'articles.UserID', '=', 'users.UserID')
            ->leftJoin('likes', 'articles.ArticleID', '=', 'likes.ArticleID')
            ->whereNotNull('races.RaceDateTime')
            ->groupBy(
                'articles.ArticleID',
                'races.RaceDateTime',
                'articles.Title',
                'articles.Price',
                'articles.PublishedAt',
                'users.IconImageURL',
                'users.name',
                'articles.TopImageURL'
            )
            ->orderByDesc('races.RaceDateTime')
            ->orderByDesc('articles.PublishedAt')
            ->limit(4)
            ->get()
            ->toArray();
    }
}

<?php

namespace App\Http\Controllers;

use App\Http\Requests\ProfileUpdateRequest;
use Illuminate\Http\RedirectResponse;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;
use Illuminate\Support\Facades\Redirect;
use Illuminate\View\View;

class ProfileController extends Controller
{
    /**
     * Display the user's profile form.
     */
    public function edit(Request $request): View
    {
        return view('profile.edit', [
            'user' => $request->user(),
        ]);
    }

    /**
     * Update the user's profile information.
     */
    public function update(ProfileUpdateRequest $request): RedirectResponse
    {
        $request->user()->fill($request->validated());

        if ($request->user()->isDirty('email')) {
            $request->user()->email_verified_at = null;
        }

        $request->user()->save();

        return Redirect::route('profile.edit')->with('status', 'profile-updated');
    }

    /**
     * Delete the user's account.
     */
    public function destroy(Request $request): RedirectResponse
    {
        $request->validateWithBag('userDeletion', [
            'password' => ['required', 'current_password'],
        ]);

        $user = $request->user();

        Auth::logout();

        $user->delete();

        $request->session()->invalidate();
        $request->session()->regenerateToken();

        return Redirect::to('/');
    }
}
